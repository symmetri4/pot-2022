--- SQL schema ---

CREATE TABLE if not exists Participants (
    identifier INTEGER PRIMARY KEY,
    trial_no INTEGER /* sanity check */
);

CREATE TABLE if not exists Tasks (
    id INTEGER PRIMARY KEY,
    participant_id INTEGER REFERENCES Participants,
    task_no INTEGER,
    success BOOLEAN,
    time_elapsed FLOAT
);

CREATE TABLE if not exists LoadNasa (
    id INTEGER PRIMARY KEY,
    task_no INTEGER,
    participant_id INTEGER REFERENCES Participants,
    mental_demand INTEGER,
    physical_demand INTEGER,
    temporal_demand INTEGER,
    performance INTEGER,
    effort INTEGER,
    frustration INTEGER
);

/* Filled in post-hoc from GazePoint data */
CREATE TABLE if not exists LoadPupil (
    id INTEGER PRIMARY KEY,
    identifier INTEGER,
    task_no INTEGER,
    load_score INTEGER
);

CREATE TABLE if not exists Antisaccade (
    id INTEGER PRIMARY KEY,
    participant_id INTEGER REFERENCES Participants,
    item_no INTEGER,
    success BOOLEAN
);

CREATE TABLE if not exists  VisualArrays (
    id INTEGER PRIMARY KEY,
    participant_id INTEGER REFERENCES Participants,
    item_no INTEGER,
    success BOOLEAN
);

CREATE TABLE if not exists Flanker (
    id INTEGER PRIMARY KEY,
    participant_id INTEGER REFERENCES Participants,
    item_no INTEGER,
    success BOOLEAN
);

CREATE TABLE if not exists Stroop (
    id INTEGER PRIMARY KEY,
    participant_id INTEGER REFERENCES Participants,
    item_no INTEGER,
    success BOOLEAN
);

/* Filled in post-hoc from UoH data */
CREATE TABLE if not exists Wais (
    id INTEGER PRIMARY KEY,
    identifier INTEGER,
    vci INTEGER,
    poi INTEGER,
    wmi INTEGER,
    psi INTEGER,
    verbal INTEGER,
    performance INTEGER,
    full_scale INTEGER
);
